CREATE DATABASE goobam;

ALTER DATABASE goodbam DEFAULT CHARACTER SET utf8 COLLATE UTF8_GENERAL_CI;

CREATE TABLE MAMBER (
	userID VARCHAR(100) NOT NULL,
	userPass VARCHAR(100) NOT NULL,
	userName VARCHAR(100)NOT NULL,
	userEmail VARCHAR(100),
	userQue VARCHAR(100) NOT NULL,
	userAns VARCHAR(100)NOT NULL,
	userSignupdate DATETIME NOT NULL DEFAULT NOW(),
	PRIMARY KEY (userID)
);

CREATE TABLE TEMP (
	userID VARCHAR(100) NOT NULL,
	userName VARCHAR(100)NOT NULL,
	userTemp VARCHAR(100)NOT NULL,
	uestime DATETIME NOT NULL DEFAULT NOW(),
	PRIMARY KEY (userID)
);

CREATE TABLE HUMIDTY (
	userID VARCHAR(100) NOT NULL,
	userName VARCHAR(100)NOT NULL,
	userHum VARCHAR(100)NOT NULL,
	uestime DATETIME NOT NULL DEFAULT NOW(),
	PRIMARY KEY (userID)
);
